70  /** Greedy assign (source, target) to a machine using:
71  * bitset<MAX_MACHINE> src_degree : the degree presence of source over machines
72  * bitset<MAX_MACHINE> dst_degree : the degree presence of target over machines
73  * vector<size_t> proc_num_edges : the edge counts over machines
74  * */
75  procid_t edge_to_proc_greedy (const vertex_id_type source,
76  const vertex_id_type target,
77  bin_counts_type& src_degree,
78  bin_counts_type& dst_degree,
79  std::vector<size_t>& proc_num_edges,
80  bool usehash = false,
81  bool userecent = false) {
82  size_t numprocs = proc_num_edges.size();
83 
84  // Compute the score of each proc.
85  procid_t best_proc = -1;
86  double maxscore = 0.0;
87  double epsilon = 1.0;
88  std::vector<double> proc_score(numprocs);
89  size_t minedges = *std::min_element(proc_num_edges.begin(), proc_num_edges.end());
90  size_t maxedges = *std::max_element(proc_num_edges.begin(), proc_num_edges.end());
91 
92  for (size_t i = 0; i < numprocs; ++i) {
93  size_t sd = src_degree.get(i) + (usehash && (source % numprocs == i));
94  size_t td = dst_degree.get(i) + (usehash && (target % numprocs == i));
95  double bal = (maxedges - proc_num_edges[i])/(epsilon + maxedges - minedges);
96  proc_score[i] = bal + ((sd > 0) + (td > 0));
97  }
98  maxscore = *std::max_element(proc_score.begin(), proc_score.end());
99 
100  std::vector<procid_t> top_procs;
101  for (size_t i = 0; i < numprocs; ++i)
102  if (std::fabs(proc_score[i] - maxscore) < 1e-5)
103  top_procs.push_back(i);
104 
105  // Hash the edge to one of the best procs.
106  typedef std::pair<vertex_id_type, vertex_id_type> edge_pair_type;
107  const edge_pair_type edge_pair(std::min(source, target),
108  std::max(source, target));
109  best_proc = top_procs[graph_hash::hash_edge(edge_pair) % top_procs.size()];
110 
111  ASSERT_LT(best_proc, numprocs);
112  if (userecent) {
113  src_degree.clear();
114  dst_degree.clear();
115  }
116  src_degree.set_bit(best_proc);
117  dst_degree.set_bit(best_proc);
118  ++proc_num_edges[best_proc];
119  return best_proc;
120  };

177  /** HDRF greedy assign (source, target) to a machine using:
178  * bitset<MAX_MACHINE> src_degree : the degree presence of source over machines
179  * bitset<MAX_MACHINE> dst_degree : the degree presence of target over machines
180  * size_t src_true_degree : the degree of source vertex over machines
181  * size_t dst_true_degree : the degree of target vertex over machines
182  * vector<size_t> proc_num_edges : the edge counts over machines
183  *
184  * author : Fabio Petroni [www.fabiopetroni.com]
185  * Giorgio Iacoboni [g.iacoboni@gmail.com]
186  *
187  * Based on the publication:
188  * F. Petroni, L. Querzoni, K. Daudjee, S. Kamali and G. Iacoboni:
189  * "HDRF: Stream-Based Partitioning for Power-Law Graphs".
190  * CIKM, 2015.
191  * */
192  procid_t edge_to_proc_hdrf (const vertex_id_type source,
193  const vertex_id_type target,
194  bin_counts_type& src_degree,
195  bin_counts_type& dst_degree,
196  size_t& src_true_degree,
197  size_t& dst_true_degree,
198  std::vector<size_t>& proc_num_edges,
199  bool usehash = false,
200  bool userecent = false) {
201 
202  size_t numprocs = proc_num_edges.size();
203 
204  size_t degree_u = src_true_degree;
205  degree_u = degree_u +1;
206  size_t degree_v = dst_true_degree;
207  degree_v = degree_v +1;
208  size_t SUM = degree_u + degree_v;
209  double fu = degree_u;
210  fu /= SUM;
211  double fv = degree_v;
212  fv /= SUM;
213 
214  // Compute the score of each proc.
215  procid_t best_proc = -1;
216  double maxscore = 0.0;
217  double epsilon = 1.0;
218  std::vector<double> proc_score(numprocs);
219  size_t minedges = *std::min_element(proc_num_edges.begin(), proc_num_edges.end());
220  size_t maxedges = *std::max_element(proc_num_edges.begin(), proc_num_edges.end());
221 
222  for (size_t i = 0; i < numprocs; ++i) {
223  double new_sd = 0;
224  double new_td = 0;
225  size_t sd = src_degree.get(i) + (usehash && (source % numprocs == i));
226  size_t td = dst_degree.get(i) + (usehash && (target % numprocs == i));
227  if (sd > 0){
228  new_sd = 1+(1-fu);
229  }
230  if (td > 0){
231  new_td = 1+(1-fv);
232  }
233  double bal = (maxedges - proc_num_edges[i])/(epsilon + maxedges - minedges);
234 
235  proc_score[i] = bal + new_sd + new_td;
236  }
237 
238  maxscore = *std::max_element(proc_score.begin(), proc_score.end());
239 
240  std::vector<procid_t> top_procs;
241  for (size_t i = 0; i < numprocs; ++i)
242  if (std::fabs(proc_score[i] - maxscore) < 1e-5)
243  top_procs.push_back(i);
244 
245  // Hash the edge to one of the best procs.
246  typedef std::pair<vertex_id_type, vertex_id_type> edge_pair_type;
247  const edge_pair_type edge_pair(std::min(source, target), std::max(source, target));
248  best_proc = top_procs[graph_hash::hash_edge(edge_pair) % top_procs.size()];
249 
250  ASSERT_LT(best_proc, numprocs);
251  if (userecent) {
252  src_degree.clear();
253  dst_degree.clear();
254  }
255  src_degree.set_bit(best_proc);
256  dst_degree.set_bit(best_proc);
257  ++proc_num_edges[best_proc];
258  ++src_true_degree;
259  ++dst_true_degree;
260  return best_proc;
261  };
